WITH POH AS (
    SELECT 
        poh.MicrossSalesOrderReference,

        -- ? Working days
        DATEDIFF(DAY, poh.DateRecieved, GETDATE())
        - (DATEDIFF(WEEK, poh.DateRecieved, GETDATE()) * 2)
        - CASE WHEN DATENAME(WEEKDAY, poh.DateRecieved) = 'Sunday' THEN 1 ELSE 0 END
        - CASE WHEN DATENAME(WEEKDAY, GETDATE()) = 'Saturday' THEN 1 ELSE 0 END
        AS WorkingDays

    FROM dbo.PurchaseOrderHeader poh
    WHERE 
        poh.MicrossSalesOrderReference IS NOT NULL
        AND poh.CRStatus <> 'Completed'


 
)

SELECT 
    CASE 
        WHEN WorkingDays > 10 THEN 'Overdue'
        ELSE 'On Time'
    END AS Status,
    COUNT(*) AS Count
FROM POH
GROUP BY 
    CASE 
        WHEN WorkingDays > 10 THEN 'Overdue'
        ELSE 'On Time'
    END;
