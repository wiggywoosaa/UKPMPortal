SELECT
    a.MicrossSalesOrderReference,
    a.QuestionId,
    q.Question,
    a.ResponsiblePerson,
    a.Action,
    a.DueDate,
    a.AnsweredBy,
    a.Status,
    a.ParentActionId,
    a.id
FROM dbo.POH_Answers a
JOIN dbo.Admin_Questions q
    ON a.QuestionId = q.id
WHERE
    a.IsLatest = 1
    AND a.YesNo = 0
    AND a.DueDate IS NOT NULL
    AND a.MicrossSalesOrderReference = {{ Select_Archive.selectedOptionValue }}
ORDER BY a.DueDate ASC;
