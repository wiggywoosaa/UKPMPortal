SELECT        dbo.Admin_Questions.id,dbo.Admin_Questions.Department, dbo.Admin_Questions.Question, dbo.POH_Answers.Answer, dbo.POH_Answers.YesNo, dbo.POH_Answers.AnsweredBy, dbo.POH_Answers.AnsweredDate, 
                         dbo.POH_Answers.ResponsiblePerson, dbo.POH_Answers.Action, dbo.POH_Answers.DueDate,  dbo.POH_Answers.Status, dbo.POH_Answers.IsLatest, dbo.Admin_Questions.RequiresPM,dbo.POH_Answers.MicrossSalesOrderReference,dbo.POH_Answers.QuestionId
FROM            dbo.Admin_Questions INNER JOIN
                         dbo.POH_Answers ON dbo.Admin_Questions.id = dbo.POH_Answers.QuestionId
WHERE        (dbo.Admin_Questions.RequiresPM = 1) and (dbo.POH_Answers.IsLatest = 'True') and  MicrossSalesOrderReference = {{ Select_POPM.selectedOptionValue }}
ORDER BY dbo.POH_Answers.id DESC
